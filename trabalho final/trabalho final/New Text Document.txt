DROP DATABASE IF EXISTS controle_estoque;

CREATE DATABASE controle_estoque;

USE controle_estoque;

CREATE TABLE cliente (
cliente_id INT PRIMARY KEY AUTO_INCREMENT,
nome VARCHAR(50) NOT NULL,
cpf_cnpj VARCHAR(20) NOT NULL,
telefone VARCHAR(15) NOT NULL,
email VARCHAR(100) NOT NULL,
endereco VARCHAR(100) NOT NULL,
cidade VARCHAR(50) NOT NULL,
estado VARCHAR(30) NOT NULL,
data_cadastro DATETIME DEFAULT CURRENT_TIMESTAMP,
limite_credito DECIMAL(10,2) NOT NULL,
status VARCHAR(20) NOT NULL
);

CREATE TABLE fornecedor (
fornecedor_id INT PRIMARY KEY AUTO_INCREMENT,
nome VARCHAR(50) NOT NULL,
cpf_cnpj VARCHAR(20) NOT NULL,
telefone VARCHAR(20) NOT NULL,
email VARCHAR(50) NOT NULL,
endereco VARCHAR(100) NOT NULL,
cidade VARCHAR(50) NOT NULL,
estado VARCHAR(30) NOT NULL,
data_cadastro DATETIME DEFAULT CURRENT_TIMESTAMP,
status VARCHAR(20) NOT NULL
);

CREATE TABLE categoria (
categoria_id INT PRIMARY KEY AUTO_INCREMENT,
nome VARCHAR(50) NOT NULL,
descricao VARCHAR(100) NOT NULL,
status VARCHAR(20) NOT NULL
);

CREATE TABLE produto (
produto_id INT PRIMARY KEY AUTO_INCREMENT,
codigo VARCHAR(30) NOT NULL,
descricao VARCHAR(50) NOT NULL,
preco_venda DECIMAL(10,2) NOT NULL,
preco_custo DECIMAL(10,2) NOT NULL,
estoque_atual INT NOT NULL,
estoque_minimo INT NOT NULL,
categoria_id INT,
unidade_medida VARCHAR(20) NOT NULL,
data_cadastro DATETIME DEFAULT CURRENT_TIMESTAMP,
status VARCHAR(20) NOT NULL,
FOREIGN KEY (categoria_id) REFERENCES categoria(categoria_id)
);

CREATE TABLE funcionario (
funcionario_id INT PRIMARY KEY AUTO_INCREMENT,
nome VARCHAR(50) NOT NULL,
cpf VARCHAR(20) NOT NULL,
cargo VARCHAR(50) NOT NULL,
data_contratacao DATETIME DEFAULT CURRENT_TIMESTAMP,
salario DECIMAL(10,2) NOT NULL,
telefone VARCHAR(15) NOT NULL,
email VARCHAR(60) NOT NULL,
status VARCHAR(20) NOT NULL
);

CREATE TABLE transportadora (
transportadora_id INT PRIMARY KEY AUTO_INCREMENT,
nome VARCHAR(50) NOT NULL,
cnpj VARCHAR(20) NOT NULL,
telefone VARCHAR(20) NOT NULL,
email VARCHAR(50) NOT NULL,
status VARCHAR(20) NOT NULL
);

CREATE TABLE rota (
rota_id INT PRIMARY KEY AUTO_INCREMENT,
nome VARCHAR(50) NOT NULL,
origem VARCHAR(50) NOT NULL,
destino VARCHAR(50) NOT NULL,
distancia_km DECIMAL(10,2) NOT NULL,
tempo_estimado INT NOT NULL,
transportadora_id INT,
status VARCHAR(20) NOT NULL,
FOREIGN KEY (transportadora_id) REFERENCES transportadora(transportadora_id)
);

CREATE TABLE venda (
venda_id INT PRIMARY KEY AUTO_INCREMENT,
data_venda DATETIME DEFAULT CURRENT_TIMESTAMP,
cliente_id INT NOT NULL,
funcionario_id INT NOT NULL,
valor_total DECIMAL(10,2) NOT NULL,
desconto DECIMAL(10,2) NOT NULL,
forma_pagamento VARCHAR(20) NOT NULL,
status VARCHAR(20) NOT NULL,
observacao VARCHAR(100),
FOREIGN KEY (cliente_id) REFERENCES cliente(cliente_id),
FOREIGN KEY (funcionario_id) REFERENCES funcionario(funcionario_id)
);

CREATE TABLE item_venda (
item_venda_id INT PRIMARY KEY AUTO_INCREMENT,
venda_id INT NOT NULL,
produto_id INT NOT NULL,
quantidade INT NOT NULL,
preco_unitario DECIMAL(10,2) NOT NULL,
desconto DECIMAL(10,2) NOT NULL,
valor_total DECIMAL(10,2) NOT NULL,
FOREIGN KEY (venda_id) REFERENCES venda(venda_id),
FOREIGN KEY (produto_id) REFERENCES produto(produto_id)
);

CREATE TABLE compra (
compra_id INT PRIMARY KEY AUTO_INCREMENT,
data_compra DATETIME DEFAULT CURRENT_TIMESTAMP,
fornecedor_id INT NOT NULL,
funcionario_id INT NOT NULL,
valor_total DECIMAL(10,2) NOT NULL,
status VARCHAR(20) NOT NULL,
observacao VARCHAR(100),
FOREIGN KEY (fornecedor_id) REFERENCES fornecedor(fornecedor_id),
FOREIGN KEY (funcionario_id) REFERENCES funcionario(funcionario_id)
);

CREATE TABLE item_compra (
item_compra_id INT PRIMARY KEY AUTO_INCREMENT,
compra_id INT NOT NULL,
produto_id INT NOT NULL,
quantidade INT NOT NULL,
preco_unitario DECIMAL(10,2) NOT NULL,
valor_total DECIMAL(10,2) NOT NULL,
FOREIGN KEY (compra_id) REFERENCES compra(compra_id),
FOREIGN KEY (produto_id) REFERENCES produto(produto_id)
);

CREATE TABLE producao (
producao_id INT PRIMARY KEY AUTO_INCREMENT,
data_inicio DATETIME DEFAULT CURRENT_TIMESTAMP,
data_fim DATETIME,
funcionario_id INT,
status VARCHAR(20) NOT NULL,
observacao VARCHAR(100),
FOREIGN KEY (funcionario_id) REFERENCES funcionario(funcionario_id)
);

CREATE TABLE item_producao (
item_producao_id INT PRIMARY KEY AUTO_INCREMENT,
producao_id INT NOT NULL,
produto_id INT NOT NULL,
quantidade_planejada INT NOT NULL,
quantidade_produzida INT NOT NULL,
custo_unitario DECIMAL(10,2) NOT NULL,
valor_total DECIMAL(10,2) NOT NULL,
FOREIGN KEY (producao_id) REFERENCES producao(producao_id),
FOREIGN KEY (produto_id) REFERENCES produto(produto_id)
);

CREATE TABLE pedido (
pedido_id INT PRIMARY KEY AUTO_INCREMENT,
data_pedido DATETIME DEFAULT CURRENT_TIMESTAMP,
cliente_id INT NOT NULL,
funcionario_id INT NOT NULL,
data_entrega_prevista DATE NOT NULL,
rota_id INT,
valor_total DECIMAL(10,2) NOT NULL,
status VARCHAR(20) NOT NULL,
observacao VARCHAR(100),
FOREIGN KEY (cliente_id) REFERENCES cliente(cliente_id),
FOREIGN KEY (funcionario_id) REFERENCES funcionario(funcionario_id),
FOREIGN KEY (rota_id) REFERENCES rota(rota_id)
);

CREATE TABLE item_pedido (
item_pedido_id INT PRIMARY KEY AUTO_INCREMENT,
pedido_id INT NOT NULL,
produto_id INT NOT NULL,
quantidade INT NOT NULL,
preco_unitario DECIMAL(10,2) NOT NULL,
desconto DECIMAL(10,2) NOT NULL,
valor_total DECIMAL(10,2) NOT NULL,
FOREIGN KEY (pedido_id) REFERENCES pedido(pedido_id),
FOREIGN KEY (produto_id) REFERENCES produto(produto_id)
);

SELECT t.nome AS transportadora,pr.descricao AS produto,SUM(ip.quantidade) AS quantidade_vendida
FROM transportadora t
INNER JOIN rota r
    ON t.transportadora_id = r.transportadora_id
INNER JOIN pedido pe
    ON r.rota_id = pe.rota_id
INNER JOIN item_pedido ip
    ON pe.pedido_id = ip.pedido_id
INNER JOIN produto pr
    ON ip.produto_id = pr.produto_id
GROUP BY t.transportadora_id,t.nome,pr.produto_id,pr.descricao
ORDER BY t.nome,quantidade_vendida DESC;

SELECT pe.pedido_id,pe.data_pedido,c.nome AS cliente,SUM(ip.valor_total) AS total_dos_itens
FROM pedido pe
INNER JOIN cliente c
    ON pe.cliente_id = c.cliente_id
INNER JOIN item_pedido ip
    ON pe.pedido_id = ip.pedido_id
GROUP BY pe.pedido_id,pe.data_pedido,c.nome
HAVING SUM(ip.valor_total) > (
    SELECT AVG(valor_total)
    FROM pedido

SELECT f.funcionario_id,f.nome AS funcionario,p.producao_id,p.data_inicio,SUM(ip.valor_total) AS custo_total_producao
FROM producao p
INNER JOIN funcionario f
    ON p.funcionario_id = f.funcionario_id
INNER JOIN item_producao ip
    ON p.producao_id = ip.producao_id
GROUP BY f.funcionario_id,f.nome,p.producao_id,p.data_inicio
HAVING SUM(ip.valor_total) > 10000
ORDER BY f.nome,custo_total_producao DESC;







SELECT 
    c.cliente_id,
    c.nome AS cliente,
    pe.pedido_id,
    pe.valor_total,
    COUNT(DISTINCT ip.produto_id) AS produtos_diferentes
FROM cliente c
INNER JOIN pedido pe
    ON c.cliente_id = pe.cliente_id
INNER JOIN item_pedido ip
    ON pe.pedido_id = ip.pedido_id
CROSS JOIN (
    SELECT AVG(valor_total) AS media_pedidos
    FROM pedido
) AS media
GROUP BY 
    c.cliente_id,
    c.nome,
    pe.pedido_id,
    pe.valor_total,
    media.media_pedidos
HAVING 
    COUNT(DISTINCT ip.produto_id) > 3
    AND pe.valor_total > media.media_pedidos;
